-- AND OR NOT -- Logical Operatore 

SELECT *
FROM employee_demographics
WHERE gender = 'Male'
AND birth_date > '1985-01-01' ;

SELECT *
FROM employee_demographics
WHERE gender = 'Male'
OR birth_date > '1985-01-01' ;

SELECT *
FROM employee_demographics
WHERE gender = 'Male'
OR NOT birth_date > '1985-01-01' ;

SELECT *
FROM employee_demographics
WHERE gender = 'Male'
OR NOT (first_name = 'leslie' AND age = 44) OR age > 55;
